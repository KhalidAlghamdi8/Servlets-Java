package com.examples.lesson06.services;

import java.text.SimpleDateFormat;
import java.util.Date;
import javax.enterprise.context.ApplicationScoped;

@ApplicationScoped
public class InfoServiceImpl implements InfoBean {

  @Override
  public String calculateServerName() {
    return "TEST SERVER";
  }

  @Override
  public String calculateServerMemory() {
    return formatBytes(Runtime.getRuntime().maxMemory());
  }

  @Override
  public String calculateServerFreeMemory() {
    return formatBytes(Runtime.getRuntime().freeMemory());
  }

  @Override
  public String calculateServerCores() {
    return Integer.toString(Runtime.getRuntime().availableProcessors());
  }

  @Override
  public String calculateServerTime() {
    return new SimpleDateFormat().format(new Date());
  }

  private String formatBytes(long bytes) {
    if (bytes < 1024) {
      return bytes + "B";
    }
    if (bytes < 1024 * 1024) {
      return (bytes / 1024) + "KB";
    }
    return (bytes / (1024 * 1024)) + "MB";
  }
}
This is the implementation of the InfoService interface.
The applicationScoped annotation tells the Server that it should instantiate this bean and make it avaliable to all the injecting objects.
This annotation also specifies the scope of the variable. ApplicationScope objects are instantiated only once per application meaning the same object is shared by every object in the application.
The implementation is a simple JavaBeans object.
You can refer to CDI and JavaBeans documentation for additional features and annotations available to configure beans.
Some available annotations you may use are:
  • @Inject: Injects another bean.
  • @PostContruct: Executes the method after the constructor and injection occurs.
  • @PreDestroy: Executes the method before the object is disposed (goes out of scope)
