package com.examples.lesson04;

import java.io.IOException;
import java.io.PrintWriter;
import javax.servlet.ServletException;
import javax.servlet.annotation.WebServlet;
import javax.servlet.http.HttpServlet;
import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;
import javax.servlet.http.HttpSession;

@WebServlet(name = "GreetingServlet", urlPatterns = {"/greet"})
public class GreetingServlet extends HttpServlet {
  
  protected void processRequest(HttpServletRequest request, HttpServletResponse response)
          throws ServletException, IOException {
    HttpSession session = request.getSession();
    String username = (String) session.getAttribute("user_name");
    String role = (String) session.getAttribute("user_role");
    
    if (role == null || username == null) {
      response.sendRedirect("loginForm.html");
    } else {
      response.setContentType("text/html;charset=UTF-8");
      PrintWriter out = response.getWriter();
      out.println("<html>");
      out.println("<head>");
      out.println("<title>Greetings!</title>");
      out.println("<link rel=\"stylesheet\" href=\"res/styles.css\" type=\"text/css\"/>");
      out.println("</head>");
      out.println("<body>");
      
      out.println("<h1>Hello " + username + "</h1>");
      out.println("<h3>You are a logged in as a " + role + "</h2>");
      
      out.println("<a href=\"index.html\">Go Home</a>");
      out.println("</body>");
      out.println("</html>");
    }
  }

  // <editor-fold defaultstate="collapsed" desc="HttpServlet methods. Click on the + sign on the left to edit the code.">
  /**
   * Handles the HTTP
   * <code>GET</code> method.
   *
   * @param request servlet request
   * @param response servlet response
   * @throws ServletException if a servlet-specific error occurs
   * @throws IOException if an I/O error occurs
   */
  @Override
  protected void doGet(HttpServletRequest request, HttpServletResponse response)
          throws ServletException, IOException {
    processRequest(request, response);
  }

  /**
   * Handles the HTTP
   * <code>POST</code> method.
   *
   * @param request servlet request
   * @param response servlet response
   * @throws ServletException if a servlet-specific error occurs
   * @throws IOException if an I/O error occurs
   */
  @Override
  protected void doPost(HttpServletRequest request, HttpServletResponse response)
          throws ServletException, IOException {
    processRequest(request, response);
  }

  /**
   * Returns a short description of the servlet.
   *
   * @return a String containing servlet description
   */
  @Override
  public String getServletInfo() {
    return "Short description";
  }// </editor-fold>
}
This servlet is the entry point for the application.
It determines if the user is logged in or not.
If the user is logged in, a greeting is displayed.
Else, the user is redirected to the login form where and asked for credentials.
This is a simple approach to an Model-View-Controller (MVC) pattern.
Instead of displaying the entire login form, the Servlet redirects to another page to display the form.
Since the form is only HTML, the servlet redirects to an HTML page.
The request.getSession() method returns the session for the user.
HttpSession live for as long as the user keeps the browser window open.
A session may be terminated if it expires after the timeout. The session timeout is configurable in the web.xml file.
Here is an example of how to configure session timeouts: 
    <session-config>
        <session-timeout>30</session-timeout>
    </session-config>
You can store and get attributes in the session.
A session can store any type of objects inside as a key/value store.
Note that since anything can be stored inside a session, the attributes to the desired class must be cast.
Session attributes return null if they are empty or not set.
In this case if the session attributes are not set, the page will redirect the user to the login form. This is done by a simple redirect that tells the browser to navigate to the page. 
  response.sendRedirect("loginForm.html");
if the values are set then they are displayed on screen along with a greeting to the user.
