package com.examples.lesson07.view;

import com.examples.lesson07.model.Product;
import com.examples.lesson07.model.ProductCategory;
import com.examples.lesson07.model.ShoppingCart;
import java.util.ArrayList;
import java.util.List;

public class ProductCategoryView {

  private final String name;
  private final List<ProductView> products;

  public ProductCategoryView(ProductCategory category, ShoppingCart shoppingCart) {
    this.name = category.getName();
    this.products = new ArrayList<ProductView>();
    for (Product product : category.getProducts()) {
      this.products.add(new ProductView(product, shoppingCart.getQuantity(product.getId())));
    }
  }

  public String getName() {
    return name;
  }

  public List<ProductView> getProducts() {
    return products;
  }
}
This class contains the category name and a list of ProductView objects. Instances of this class are meant to be used in the JSP page for display.
This class is almost identical to the ProductCategory class, except that it contains a list of ProductView objects instead of Product objects.
At the moment, the class takes all the products inside the category and "transforms" them into ProductView objects.
